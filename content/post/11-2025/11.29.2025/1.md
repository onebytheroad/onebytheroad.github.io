#include <stdio.h>
#include <string.h>
#include <malloc.h>

void Jiemi(short* num_arr,char len,char* flag)
{
	int num = 16;
	num_arr = num - 2;
	for(int i=0;i<len&&num>=0;i++)
	{
		short char1 = *num_arr;
		char1 = (char1 - 0x502) / 2;

		char1 = char1 ^ 0x1f6e;

		char char_high = (char1 >> 8) & 0x00ff;

		char char_low = char1 & 0x00ff;

		flag[i*2] = char_high;

		flag[i*2+1] = char_low;


		num_arr = num_arr - 1;
	}

}

int main()
{
	printf("input str:");

	char* num_arr = (char*)malloc(32 * sizeof(char));

	scanf_s("%32s", num_arr,32);

	char* flag = (char*)malloc(32 * sizeof(char));

	char num = strlen(num_arr);

	Jiemi(num_arr,num,flag);
	
	free(num_arr);
	free(flag);
	

}

#include <stdlib.h> 
#include <stdbool.h>

int main(void)
{
    int a = 10;

    int* p = &a;
    int** pp = &p;

    printf("a:%d\n", a);//10
    printf("&a:%p\n", &a);//addr of a
    printf("p:%p\n", p);//addr of a
    printf("&p:%p\n", &p);//addr of p
    printf("pp:%p\n",pp);//addr of p
    printf("*p:%d\n",*p);//10
    printf("*pp:%p\n", *pp);//p,addr of a
    printf("**pp:%d\n", **pp);//10

    return 0;
}

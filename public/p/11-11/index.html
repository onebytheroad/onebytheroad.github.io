	if (err != 0 || fp == NULL)
	{
		return -1;

	}

	optwrite_record(fp);

	fclose(fp);

	err = fopen_s(&fp, FILENAME, "r");
	if (err != 0 || fp == NULL)
	{
		return -1;
	}
	
	optread_record(fp);

	optquery_record(fp);

	fclose(fp);

	return 0;

char* my_itoa(int value, char* str, int base)
{
	if (base < 2 || base > 32)
	{
		printf("Wrong radix!\n");
		return str;
	}

	if (value == 0)
	{
		*str++ = '0';
		*str = '\0';
		return ret;
	}
	if (base == 10 && value < 0)
	{
		value = -value;
		*str++ = '-';
	}
	char* start = str;
	// ä»å³åˆ°å·¦ä¾æ¬¡å°†æ•°å­—çš„æ¯ä¸€ä½å­˜å‚¨èµ·æ¥

		{
			*str++ = '0' + (char)(num % base);
		}
		else
		{
			*str++ = 'a' + (char)(num % base - 10);
		}
		num /= base;
	}
	*str = '\0';
	// å€’ç½®å­—ç¬¦ä¸²
	for (char* left = start, *right = str - 1; left < right; left++, right--)
	{
		char tmp = *left;

		*right = tmp;
	}
	return ret;
}
